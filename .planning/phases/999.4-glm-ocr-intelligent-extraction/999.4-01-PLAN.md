import pytest
from unittest.mock import MagicMock
from carta.vision.classifier import ContentClassifier, ContentType


class TestContentClassifierTextPages:
    """Pages dominated by text should classify as TEXT."""
    
    def test_pure_text_page(self):
        """Page with only paragraphs → TEXT."""
        classifier = ContentClassifier()
        mock_page = MagicMock()
        mock_page.rect.width = 612  # Letter size
        mock_page.rect.height = 792
        
        # Simulate text covering 80% of page
        mock_page.get_text.return_value = [{
            'type': 0, 'x0': 50, 'y0': 50, 'x1': 562, 'y1': 100  # 512x50 block
        }] * 10  # 10 text blocks
        
        result = classifier.classify_page(mock_page)
        assert result.content_type == ContentType.TEXT
        assert result.text_coverage > 0.70
    
    def test_table_page(self):
        """Page with structured tables → TEXT."""
        classifier = ContentClassifier()
        mock_page = MagicMock()
        
        # Low text coverage but table patterns detected
        mock_page.get_text.return_value = []
        mock_page.get_images.return_value = []
        
        with patch.object(classifier, '_detect_table_patterns', return_value=True):
            result = classifier.classify_page(mock_page)
            assert result.content_type == ContentType.TEXT
            assert result.has_tables is True


class TestContentClassifierVisualPages:
    """Image-heavy pages should classify as VISUAL."""
    
    def test_diagram_page(self):
        """Page with schematic/diagram → VISUAL."""
        classifier = ContentClassifier()
        mock_page = MagicMock()
        mock_page.rect.width = 612
        mock_page.rect.height = 792
        
        # Minimal text, large image
        mock_page.get_text.return_value = []
        mock_page.get_images.return_value = [(0, 0, 500, 600, 8, 0, 0)]
        
        result = classifier.classify_page(mock_page)
        assert result.content_type == ContentType.VISUAL
        assert result.image_coverage > 0.40
    
    def test_plot_graph_page(self):
        """Page with plot/chart → VISUAL."""
        classifier = ContentClassifier()
        mock_page = MagicMock()
        
        # Some text (axis labels) but mostly plot area
        mock_page.get_text.return_value = [
            {'type': 0, 'x0': 50, 'y0': 50, 'x1': 100, 'y1': 20}  # Small text
        ]
        mock_page.get_images.return_value = [(0, 0, 400, 300, 8, 0, 0)]
        
        result = classifier.classify_page(mock_page)
        assert result.content_type == ContentType.VISUAL


class TestContentClassifierMixedPages:
    """Balanced pages should classify as MIXED."""
    
    def test_half_text_half_diagram(self):
        """Page with text section + diagram → MIXED."""
        classifier = ContentClassifier()
        mock_page = MagicMock()
        mock_page.rect.width = 612

        result = classifier.classify_page(mock_page)
        assert result.content_type == ContentType.MIXED


class TestContentClassifierThresholds:
    """Threshold configuration affects classification."""
    
    def test_custom_text_threshold(self):
        """Higher threshold makes classification stricter."""
        # With default 0.70, 65% text is MIXED
        classifier_default = ContentClassifier(text_threshold=0.70)
        classifier_strict = ContentClassifier(text_threshold=0.80)
        
        mock_page = MagicMock()
        mock_page.rect.width = 612
        mock_page.rect.height = 792
        mock_page.get_text.return_value = [
            {'type': 0, 'x0': 50, 'y0': 50, 'x1': 562, 'y1': 150}  # ~65% coverage
        ]
        mock_page.get_images.return_value = []
        
        result_default = classifier_default.classify_page(mock_page)
        result_strict = classifier_strict.classify_page(mock_page)
        
        assert result_default.content_type == ContentType.MIXED
        assert result_strict.content_type == ContentType.MIXED  # Still not TEXT


class TestTableDetection:
    """Table pattern detection."""
    
    def test_detects_markdown_like_tables(self):
        """Structural patterns → table detected."""
        classifier = ContentClassifier()
        
        # Simulate aligned text columns (table-like)
        text_blocks = [
            {'type': 0, 'x0': 50, 'y0': 100, 'x1': 150, 'y1': 20},   # Col 1
            {'type': 0, 'x0': 200, 'y0': 100, 'x1': 300, 'y1': 20},  # Col 2
            {'type': 0, 'x0': 350, 'y0': 100, 'x1': 450, 'y1': 20},  # Col 3
            {'type': 0, 'x0': 50, 'y0': 130, 'x1': 150, 'y1': 20},   # Next row
            {'type': 0, 'x0': 200, 'y0': 130, 'x1': 300, 'y1': 20},
            {'type': 0, 'x0': 350, 'y0': 130, 'x1': 450, 'y1': 20},
        ]
        
        assert classifier._detect_table_patterns(text_blocks) is True
    
    def test_ignores_single_column_text(self):
        """Regular paragraphs not detected as tables."""
        classifier = ContentClassifier()

            {'type': 0, 'x0': 50, 'y0': 100, 'x1': 562, 'y1': 20},  # Full width
            {'type': 0, 'x0': 50, 'y0': 130, 'x1': 562, 'y1': 20},
        ]
        
        assert classifier._detect_table_patterns(text_blocks) is False

## Implementation Steps (GREEN Phase)

1. **Create module structure**
   ```
   carta/vision/
   ├── __init__.py
   ├── classifier.py          # Main implementation
   └── tests/
       └── test_classifier.py   # Tests above
   ```

2. **Implement `ContentClassifier.classify_page()`**
   - Calculate text coverage via PyMuPDF `get_text("blocks")`
   - Calculate image coverage via `get_images()`
   - Run table detection heuristics
   - Apply thresholds to determine ContentType

3. **Implement `_detect_table_patterns()`**
   - Analyze x-coordinates of text blocks
   - Detect aligned columns (table signature)
   - Return boolean

4. **Implement `classify_image()` (optional)**
   - For pre-rendered images (not PyMuPDF pages)
   - Use image entropy analysis as fallback

5. **Run tests until all pass**
