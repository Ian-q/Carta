```python
def _judge_with_timeout(prompt, hits, cfg, timeout_s) -> bool:
    with concurrent.futures.ThreadPoolExecutor(max_workers=1) as executor:
        future = executor.submit(_call_ollama_judge, prompt, hits, cfg)
        try:
            return future.result(timeout=timeout_s)
        except concurrent.futures.TimeoutError:
            print(f"carta-hook: judge timeout after {timeout_s}s (fail-open)", file=sys.stderr)
            return False   # BUG: should be True (fail-open = inject)
        except Exception as e:
            print(f"carta-hook: judge exception (fail-open): {e}", file=sys.stderr)
            return False   # Correct: non-timeout exception stays closed
```
